### 2.2 自动保存流程

```typescript
// src/stores/graph/persistenceMiddleware.ts

import { StateCreator } from 'zustand'
import { storageManager } from '@/services/storage'
import { useWorkspaceStore } from '@/stores/workspace'
import { debounce } from '@/lib/utils'

const AUTO_SAVE_DELAY = 500 // 500ms 防抖

export const persistenceMiddleware = <T extends object>(
  config: StateCreator<T>
): StateCreator<T> => (set, get, api) => {

  // 创建防抖保存函数
  const debouncedSave = debounce(async () => {
    try {
      // 1. 获取当前图状态
      const graphState = get() as any
      const nodes = graphState.nodes || []
      const edges = graphState.edges || []
      const viewport = graphState.viewport || { x: 0, y: 0, zoom: 1 }

      // 2. 获取工作空间状态
      const { currentCanvasId, canvases, userId, canvasTree } = useWorkspaceStore.getState()

      // 3. 更新当前画布
      const currentCanvas = canvases.find(c => c.id === currentCanvasId)
      if (!currentCanvas) return

      const updatedCanvas = {
        ...currentCanvas,
        graphData: {
          nodes: nodes,  // ✅ 完整的节点数组（包含所有属性）
          edges: edges,  // ✅ 完整的边数组（包含所有属性）
        },
        viewportState: viewport,
        updatedAt: new Date(),
      }

      // 4. 构造完整工作空间
      const workspace = {
        userId,
        currentCanvasId,
        canvases: canvases.map(c =>
          c.id === currentCanvasId ? updatedCanvas : c
        ),
        canvasTree,
      }

      // 5. 保存到 localStorage
      await storageManager.saveCanvas(workspace, currentCanvasId, updatedCanvas)

      console.log('[持久化] 自动保存成功', {
        canvasId: currentCanvasId,
        nodesCount: nodes.length,
        edgesCount: edges.length,
      })
    } catch (error) {
      console.error('[持久化] 自动保存失败:', error)
    }
  }, AUTO_SAVE_DELAY)

  // 包装 set 函数
  const wrappedSet: typeof set = (...args) => {
    set(...args)
    debouncedSave()  // 触发自动保存
  }

  return config(wrappedSet, get, api)
}
```

## 3. 数据序列化处理

### 3.1 复杂类型序列化方案

**问题**：Node/Group/Edge 包含复杂类型：
- `Date` 类型（createdAt, updatedAt）
- `CSSProperties` 类型（style）
- 嵌套对象（attributes, data.customProperties）

**解决方案**：扩展 LocalStorageAdapter 的序列化器

```typescript
// src/services/storage/adapters/LocalStorageAdapter.ts

export class LocalStorageAdapter implements StorageAdapter {
  /**
   * 递归序列化数据
   * - Date → ISO string + 类型标记
   * - CSSProperties → 保持原样
   * - 其他对象 → 递归处理
   */
  private serialize(data: unknown): string {
    return JSON.stringify(data, (key, value) => {
      // 1. 处理 Date 类型
      if (value instanceof Date) {
        return {
          __type: 'Date',
          __value: value.toISOString(),
        }
      }

      // 2. 处理 undefined（JSON.stringify 会忽略 undefined）
      if (value === undefined) {
        return { __type: 'undefined' }
      }

      // 3. 其他类型保持原样
      return value
    })
  }

  /**
   * 递归反序列化数据
   */
  private deserialize<T>(serialized: string): T {
    return JSON.parse(serialized, (key, value) => {
      // 1. 恢复 Date 类型
      if (value && value.__type === 'Date') {
        return new Date(value.__value)
      }

      // 2. 恢复 undefined
      if (value && value.__type === 'undefined') {
        return undefined
      }

      // 3. 其他类型保持原样
      return value
    })
  }

  async save(key: string, data: unknown): Promise<void> {
    try {
      const serialized = this.serialize(data)

      // 检查数据大小（localStorage 限制约 5-10MB）
      const sizeInMB = new Blob([serialized]).size / 1024 / 1024
      if (sizeInMB > 5) {
        console.warn(`[存储警告] 数据量较大 (${sizeInMB.toFixed(2)}MB)，建议清理或切换到 IndexedDB`)
      }

      localStorage.setItem(key, serialized)
    } catch (error) {
      if (error instanceof Error && error.name === 'QuotaExceededError') {
        throw new Error('存储空间不足，请清理部分数据或导出备份')
      }
      throw error
    }
  }

  async load<T>(key: string): Promise<T | null> {
    try {
      const serialized = localStorage.getItem(key)
      if (!serialized) return null
      return this.deserialize<T>(serialized)
    } catch (error) {
      console.error(`加载数据失败 [${key}]:`, error)
      return null
    }
  }

  async remove(key: string): Promise<void> {
    localStorage.removeItem(key)
  }

  async clear(): Promise<void> {
    localStorage.clear()
  }
}
```

### 3.2 数据压缩（可选优化）

如果数据量很大，可以添加压缩支持：

```typescript
import pako from 'pako'

export class LocalStorageAdapter implements StorageAdapter {
  private compress = true  // 是否启用压缩

  private serialize(data: unknown): string {
    const jsonString = JSON.stringify(data, (key, value) => {
      if (value instanceof Date) {
        return { __type: 'Date', __value: value.toISOString() }
      }
      return value
    })

    if (this.compress) {
      // 使用 pako 压缩
      const compressed = pako.deflate(jsonString)
      return btoa(String.fromCharCode(...compressed))  // Base64 编码
    }

    return jsonString
  }

  private deserialize<T>(serialized: string): T {
    let jsonString = serialized

    if (this.compress) {
      try {
        // Base64 解码 + pako 解压
        const compressed = Uint8Array.from(atob(serialized), c => c.charCodeAt(0))
        jsonString = pako.inflate(compressed, { to: 'string' })
      } catch {
        // 如果解压失败，说明是未压缩的数据（向后兼容）
        jsonString = serialized
      }
    }

    return JSON.parse(jsonString, (key, value) => {
      if (value && value.__type === 'Date') {
        return new Date(value.__value)
      }
      return value
    })
  }
}
```

### 4.2 localStorage 空间管理

**监控存储使用量：**

```typescript
// src/services/storage/StorageManager.ts

export class StorageManager {
  /**
   * 获取 localStorage 使用情况
   */
  getStorageUsage(): { used: number; total: number; percentage: number } {
    let used = 0

    // 计算已使用空间
    for (const key in localStorage) {
      if (localStorage.hasOwnProperty(key)) {
        used += localStorage[key].length + key.length
      }
    }

    // localStorage 限制约 5-10MB（因浏览器而异）
    const total = 5 * 1024 * 1024  // 假设 5MB
    const percentage = (used / total) * 100

    return {
      used: used / 1024 / 1024,  // MB
      total: total / 1024 / 1024,  // MB
      percentage,
    }
  }

  /**
   * 检查是否需要清理
   */
  shouldCleanup(): boolean {
    const { percentage } = this.getStorageUsage()
    return percentage > 80  // 超过 80% 提示清理
  }

  /**
   * 获取每个画布的数据大小
   */
  async getCanvasSizes(workspace: Workspace): Promise<Map<string, number>> {
    const sizes = new Map<string, number>()

    for (const canvas of workspace.canvases) {
      const serialized = JSON.stringify(canvas.graphData)
      const sizeInMB = new Blob([serialized]).size / 1024 / 1024
      sizes.set(canvas.id, sizeInMB)
    }

    return sizes
  }
}
```

## 5. 数据导出/导入功能

### 5.1 导出画布数据

```typescript
// src/services/storage/StorageManager.ts

export class StorageManager {
  /**
   * 导出画布数据为 JSON 文件
   */
  async exportCanvas(canvasId: string): Promise<void> {
    const workspace = await this.loadWorkspace()
    if (!workspace) return

    const canvas = workspace.canvases.find(c => c.id === canvasId)
    if (!canvas) return

    // 构造导出数据
    const exportData = {
      version: STORAGE_VERSION,
      canvas: canvas,
      exportedAt: new Date().toISOString(),
    }

    // 创建下载链接
    const blob = new Blob([JSON.stringify(exportData, null, 2)], {
      type: 'application/json',
    })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `${canvas.name}_${canvasId}.json`
    a.click()
    URL.revokeObjectURL(url)
  }

  /**
   * 导入画布数据
   */
  async importCanvas(file: File): Promise<Canvas> {
    const text = await file.text()
    const data = JSON.parse(text)

    // 验证数据格式
    const validatedCanvas = CanvasSchema.parse(data.canvas)

    // 生成新 ID（避免冲突）
    const newCanvas: Canvas = {
      ...validatedCanvas,
      id: `canvas_${nanoid()}`,
      createdAt: new Date(),
      updatedAt: new Date(),
    }

    return newCanvas
  }

  /**
   * 导出所有工作空间数据
   */
  async exportWorkspace(): Promise<void> {
    const workspace = await this.loadWorkspace()
    if (!workspace) return

    const exportData = {
      version: STORAGE_VERSION,
      workspace: workspace,
      exportedAt: new Date().toISOString(),
    }

    const blob = new Blob([JSON.stringify(exportData, null, 2)], {
      type: 'application/json',
    })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `workspace_backup_${Date.now()}.json`
    a.click()
    URL.revokeObjectURL(url)
  }
}
```

## 7. 切换到 IndexedDB 的时机

**建议切换场景：**

1. ✅ 画布数量 > 50
2. ✅ 单个画布节点数 > 500
3. ✅ localStorage 使用率 > 80%
4. ✅ 用户反馈存储不足

**IndexedDB 优势：**
- 存储空间：50MB - 数百MB（因浏览器而异）
- 支持索引查询（按画布ID快速查找）
- 支持事务和并发控制
- 异步 API，不阻塞 UI

## 8. 总结

### 8.1 设计要点

✅ **完整存储**：每个画布独立存储所有节点和边的完整属性
✅ **自动保存**：持久化中间件 + 防抖处理（500ms）
✅ **序列化处理**：自定义序列化器处理 Date 和复杂类型
✅ **性能优化**：虚拟化渲染 + 懒加载 + 分页（可选）
✅ **空间管理**：监控存储使用量 + 导出/导入功能
✅ **可扩展性**：预留 IndexedDB 适配器接口

### 8.2 数据流关键点

```
画布切换 = 保存当前 + 加载目标
         ↓
graphStore.nodes/edges ←→ canvas.graphData
         ↓
持久化中间件自动同步到 localStorage
```

### 8.3 注意事项

⚠️ **Date 类型**：需要自定义序列化/反序列化
⚠️ **数据大小**：监控 localStorage 使用量，超过 80% 提示用户
⚠️ **性能优化**：大数据量场景启用虚拟化渲染
⚠️ **数据备份**：提供导出/导入功能，防止数据丢失
