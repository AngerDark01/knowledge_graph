```typescript
// src/stores/workspace/__tests__/canvasSlice.test.ts
import { create } from 'zustand'
import { createCanvasSlice } from '../canvasSlice'

describe('canvasSlice', () => {
  it('应该能够创建画布', () => {
    const store = create(createCanvasSlice)

    const newCanvas = store.getState().createCanvas('测试画布')

    expect(newCanvas.name).toBe('测试画布')
    expect(store.getState().canvases).toHaveLength(2) // 包含默认画布
  })

  it('应该能够创建子画布', () => {
    const store = create(createCanvasSlice)
    const parentId = store.getState().canvases[0].id

    const childCanvas = store.getState().createCanvas('子画布', parentId)

    expect(childCanvas.parentId).toBe(parentId)
    expect(store.getState().canvases[0].children).toContain(childCanvas.id)
  })

  it('应该能够删除画布（递归删除子画布）', () => {
    const store = create(createCanvasSlice)
    const parent = store.getState().createCanvas('父画布')
    const child = store.getState().createCanvas('子画布', parent.id)

    store.getState().deleteCanvas(parent.id)

    expect(store.getState().canvases.find(c => c.id === parent.id)).toBeUndefined()
    expect(store.getState().canvases.find(c => c.id === child.id)).toBeUndefined()
  })

  it('应该能够切换画布', () => {
    const store = create(createCanvasSlice)
    const newCanvas = store.getState().createCanvas('新画布')

    store.getState().switchCanvas(newCanvas.id)

    expect(store.getState().currentCanvasId).toBe(newCanvas.id)
  })
})
```

## 风险管理

### 高风险项

| 风险 | 影响 | 概率 | 缓解措施 |
|------|------|------|----------|
| 文件系统权限问题 | 高 | 中 | 提前测试权限，提供降级方案 |
| 数据迁移失败 | 高 | 低 | 提供数据验证和修复工具 |
| 性能问题 | 中 | 中 | 持续性能监控，优化热点 |
| 状态同步问题 | 高 | 中 | 充分测试，添加数据校验 |

### 应急预案

1. **数据丢失**: 提供数据导出功能，定期备份
2. **功能异常**: 保留旧版本入口，可快速回退
3. **性能问题**: 提供性能模式开关，禁用部分动画
4. **存储失败**: 提供本地下载功能，手动保存数据
